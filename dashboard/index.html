/* ── Energy Engine (Eget PHP Proxy Mellemlag) ────────────────────── */ async function refreshElectricity() { const errorEl = document.getElementById('elec-error'); errorEl.style.display = 'none'; try { // VI VENDER BLIKKET INDAD: Kalder dit eget PHP-script i stedet for det eksterne API const url = 'electricity.php'; const r = await fetch(url); if (!r.ok) throw new Error('HTTP ' + r.status); const data = await r.json(); // Hvis dit PHP-script sender en fejl med (f.eks. hvis alt fejler) if (data.error) throw new Error(data.message); // Dit PHP script spytter et 'timeline' array ud const timeline = data.timeline || []; if (!timeline.length) throw new Error('Ingen data tilgængelig i cachen'); // Hent priser ud til beregninger (vi sikrer os de er tal via Float) const prices = timeline.map(h => parseFloat(h.total_price || h.spot_price)); const min = Math.min(...prices); const max = Math.max(...prices); const avg = prices.reduce((a, b) => a + b, 0) / prices.length; // Find den aktuelle time baseret på din computers ur const curHour = new Date().getHours(); // Matcher timen mod 'hour' feltet fra din PHP (f.eks. "12:00" -> 12) const curData = timeline.find(h => parseInt(h.hour) === curHour); // Opdater oversigts-tabellen document.getElementById('elec-now').textContent = curData ? parseFloat(curData.total_price || curData.spot_price).toFixed(1) + ' øre' : '—'; document.getElementById('elec-min').textContent = min.toFixed(1) + ' øre'; document.getElementById('elec-max').textContent = max.toFixed(1) + ' øre'; document.getElementById('elec-avg').textContent = avg.toFixed(1) + ' øre'; // Tegn selve 24-timers oversigten i dit flotte grid const grid = document.getElementById('elec-grid'); grid.innerHTML = ''; timeline.forEach(h => { const hHour = parseInt(h.hour); const priceVal = parseFloat(h.total_price || h.spot_price); const cell = document.createElement('div'); cell.className = 'hour-cell' + (hHour === curHour ? ' current-hour' : ''); cell.innerHTML = `
${String(hHour).padStart(2,'0')}:00
${priceVal.toFixed(0)}
`; grid.appendChild(cell); }); // Hvis din PHP sendte en statusbesked med (fx "Serving backup cache"), viser vi den i statusbaren if (data.status) { setStatusText(data.status); } } catch (e) { errorEl.style.display = 'block'; errorEl.textContent = 'El-netværksfejl: ' + e.message; setStatusText('Fejl ved synkronisering af elpriser'); } }
